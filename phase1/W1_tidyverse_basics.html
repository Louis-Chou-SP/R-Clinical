Exercise 3: Group Summary

# Group by cyl, calculate mean mpg, max hp, count of cars
result3 <- mtcars %>%
  group_by(cyl) %>%
  summarise(
    mpg_mean = mean(mpg),
    max_hp   = max(hp),
    n        = n()
  )
